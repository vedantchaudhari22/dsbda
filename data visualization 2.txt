# Step 1: Import required libraries
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Step 2: Load the Titanic dataset
titanic = sns.load_dataset('titanic')

# Step 3: Check for null values in 'age' and fill them with the mean
print("Null values before filling:\n", titanic[['age', 'sex', 'survived']].isnull().sum())

# Fill missing age values with the mean
titanic['age'].fillna(titanic['age'].mean(), inplace=True)

# Recheck if nulls remain
print("\nNull values after filling:\n", titanic[['age', 'sex', 'survived']].isnull().sum())

# Step 4: Box plot for age distribution by sex and survival
plt.figure(figsize=(10, 6))
sns.boxplot(data=titanic, x='sex', y='age', hue='survived', palette='Set2')
plt.title('Age Distribution by Gender and Survival Status')
plt.xlabel('Gender')
plt.ylabel('Age')
plt.legend(title='Survived', labels=['No (0)', 'Yes (1)'])
plt.grid(True)
plt.show()
